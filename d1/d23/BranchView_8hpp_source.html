416 static constexpr size_t function_arity = Traits::Arity;
417
418 using function_ret_type = typename Traits::ret_type;
419
420 public:
421 LambdaFieldRef(Func lambda, std::vector<FieldPtr> lambda_args) : lambda_(std::move(lambda)), lambda_args_(std::move(lambda_args)) {
422 if (function_arity != lambda_args_.size()) {
423 throw std::out_of_range("Function arity is not consistent with number of passed arguments");
424 }
425 }
426
427 double GetValue(size_t i_channel) const final {
428 return GetValueImpl(i_channel, std::make_index_sequence<function_arity>());

432 return GetValueImpl(i_channel, std::make_index_sequence<function_arity>());

437 }

442 return lambda_args_[ArgIndex]->GetValue(i_channel);
443 }
444
445 template<size_t... ArgIndex>
446 function_ret_type
447 GetValueImpl(size_t i_channel, std::index_sequence<ArgIndex...>) const {
448 return lambda_(GetArgValue<ArgIndex>(i_channel)...);

458class BranchViewDefineAction : public IAction {
459
460 class DefineActionResultImpl : public IBranchView {
461 public:
462 DefineActionResultImpl(std::string defined_field_name,
463 std::vector<std::string> lambda_args,
464 Function lambda,
465 BranchViewPtr origin) : defined_field_name_(std::move(defined_field_name)),
466 lambda_args_(std::move(lambda_args)),
467 lambda_(std::move(lambda)),
468 origin_(std::move(origin)) {
469 std::vector<FieldPtr> lambda_args_ptrs;
470 lambda_args_ptrs.reserve(lambda_args_.size());
471 for (auto& arg_field_name : lambda_args_) {
472 lambda_args_ptrs.emplace_back(origin_->GetFieldPtr(arg_field_name));
473 }
474 defined_field_ptr_ = std::make_shared<Details::LambdaFieldRef<Function>>(lambda_, lambda_args_ptrs);
475 }

480 std::vector<std::string> fields({defined_field_name_});

482 std::copy(origin_fields.begin(), origin_fields.end(), std::back_inserter(fields));
483 return fields;
484 }
485 size_t GetNumberOfChannels() const final {
486 return origin_->GetNumberOfChannels();
487 }
488 void SetEntry(Long64_t entry) const final {
489 origin_->SetEntry(entry);
490 }
491 BranchViewPtr Clone() const final {
492 return std::make_shared<DefineActionResultImpl>(defined_field_name_, lambda_args_, lambda_, origin_->Clone());
493 }
494 FieldPtr GetFieldPtr(std::string field_name) const final {
495 if (field_name == defined_field_name_) {
496 return defined_field_ptr_;
497 }
498 return origin_->GetFieldPtr(field_name);
499 }
500
501 std::string defined_field_name_;
502 std::vector<std::string> lambda_args_;
503 Function lambda_;
504 BranchViewPtr origin_;
505 FieldPtr defined_field_ptr_;
506 };
507
508 public:
509 BranchViewDefineAction(std::string field_name,
510 std::vector<std::string> lambda_args,
511 Function lambda) : defined_field_name_(std::move(field_name)),
512 lambda_args_(std::move(lambda_args)),
513 lambda_(std::move(lambda)) {
514 }
515 BranchViewPtr ApplyAction(const BranchViewPtr& origin) final {
516 /* check if all fields exist in the origin */
517 auto missing_args = Details::GetMissingArgs(lambda_args_, origin->GetFields());
518 Details::ThrowMissingArgs(missing_args);
519 /* check if new field is already defined in the origin */
520 auto origin_fields = origin->GetFields();
521 if (std::find(origin_fields.begin(), origin_fields.end(), defined_field_name_) != origin_fields.end()) {
522 throw std::runtime_error("New variable already exists in the input view");
523 }
524
525 return std::make_shared<DefineActionResultImpl>(defined_field_name_, lambda_args_, lambda_, origin);
526 }
527
528 private:
529 std::string defined_field_name_;
530 std::vector<std::string> lambda_args_;
531 Function lambda_;

534template<typename Function>
535class BranchViewFilterAction : public IAction {
536 typedef typename Details::FunctionTraits<Function>::ret_type function_ret_type;
537
538 class FilterActionResultImpl : public IBranchView {
539 struct ChannelsHolder {
540 bool is_initialized{false};
541 std::vector<size_t> channels;
542 };
543
544 typedef std::shared_ptr<ChannelsHolder> ChannelsHolderPtr;
545
546 class FieldRefImpl : public IFieldRef {
547 public:
548 FieldRefImpl(FieldPtr origin_field, const ChannelsHolderPtr& cache) : origin_field_(std::move(origin_field)), cache_(cache) {}
549 double GetValue(size_t i_channel) const override {
550 return origin_field_->GetValue(cache_->channels[i_channel]);
551 }
552 std::string GetFieldTypeStr() const override {
553 return origin_field_->GetFieldTypeStr();
554 }
555
556 private:
557 FieldPtr origin_field_;
558 ChannelsHolderPtr cache_;
559 };
560
561 public:
562 FilterActionResultImpl(Function lambda,
563 std::vector<std::string> lambda_args,
564 const BranchViewPtr& origin) : lambda_(lambda),
565 lambda_args_(std::move(lambda_args)),
566 origin_(origin) {
567 using LambdaFieldRef = Details::LambdaFieldRef<Function>;
568
569 std::vector<FieldPtr> lambda_args_ptrs;
570 lambda_args_ptrs.reserve(lambda_args_.size());
571 for (auto& arg_name : lambda_args_) {
572 lambda_args_ptrs.emplace_back(origin->GetFieldPtr(arg_name));
573 }
574 predicate_ = std::make_shared<LambdaFieldRef>(lambda_, lambda_args_ptrs);
575 cache_ = std::make_shared<ChannelsHolder>();
576 }
577 std::vector<std::string> GetFields() const override {
578 return origin_->GetFields();
579 }
580 size_t GetNumberOfChannels() const override {
581 return cache_->channels.size();
582 }
583 FieldPtr GetFieldPtr(std::string field_name) const override {
584 return std::make_shared<FieldRefImpl>(origin_->GetFieldPtr(field_name), cache_);
585 }
586 void SetEntry(Long64_t entry) const override {
587 origin_->SetEntry(entry);
588 UpdateCache();
589 }
590 Long64_t GetEntries() const override {
591 return origin_->GetEntries();
592 }
593 BranchViewPtr Clone() const override {
594 return std::make_shared<FilterActionResultImpl>(lambda_, lambda_args_, origin_->Clone());
595 }
596
597 private:
598 void UpdateCache() const {
599 cache_->is_initialized = true;
600 cache_->channels.clear();
601 cache_->channels.reserve(origin_->GetNumberOfChannels());
602 for (size_t i_channel = 0; i_channel < origin_->GetNumberOfChannels(); i_channel++) {
603 auto predicate_result = predicate_->GetValueRaw(i_channel);
604 if (predicate_result) {
605 cache_->channels.push_back(i_channel);
606 }
607 }
608 }
609
610 Function lambda_;
611 std::vector<std::string> lambda_args_;
612 BranchViewPtr origin_;
613
614 ChannelsHolderPtr cache_;
615 std::shared_ptr<Details::LambdaFieldRef<Function>> predicate_;
616 };
617
618 public:
619 BranchViewFilterAction(std::vector<std::string> lambda_args, Function lambda) : lambda_args_(std::move(lambda_args)), lambda_(std::move(lambda)) {}
620 BranchViewPtr ApplyAction(const BranchViewPtr& origin) override {
621 /* check if all fields exist in the origin */
622 auto missing_args = Details::GetMissingArgs(lambda_args_, origin->GetFields());
623 Details::ThrowMissingArgs(missing_args);
624
625 if (!std::is_same_v<bool, function_ret_type>) {
626 throw std::runtime_error("Function return type must be boolean");
627 }
628
629 return std::make_shared<FilterActionResultImpl>(lambda_, lambda_args_, origin);
630 }
631
632 private:
633 std::vector<std::string> lambda_args_;
634 Function lambda_;
635};

740class CombiningMergeAction : public IAction {
741
742 class CombiningMergeActionResultImpl : public IBranchView {
743
744 struct ChannelIndex {
745 enum ESide { LEFT,
746 RIGHT };
747 void UpdateIndex(std::size_t l, std::size_t r) {
748 left_n = l;
749 right_n = r;
750 is_initialized = true;
751 }
752 bool isValidIndex(std::size_t index) const {
753 return index < left_n * right_n;
754 }
755 std::size_t globalToLocal(std::size_t global) const {
756 if (side == LEFT) {
757 return global % left_n;
758 } else if (side == RIGHT) {
759 return global / left_n;
760 }
761 __builtin_unreachable();
762 };
763
764 bool is_initialized{false};
765 ESide side{LEFT};
766 std::size_t left_n{0};
767 std::size_t right_n{0};
768 };
769
770 class FieldImpl : public IFieldRef {
771 public:
772 FieldImpl(std::shared_ptr<const ChannelIndex> index, FieldPtr origin_field) : index_(std::move(index)), origin_field_(std::move(origin_field)) {}
773 double GetValue(size_t i_channel) const override {
774 if (index_->isValidIndex(i_channel)) {
775 return origin_field_->GetValue(index_->globalToLocal(i_channel));
776 }
777 return std::nan("<invalid-channel-id>");
778 }
779 std::string GetFieldTypeStr() const override {
780 return origin_field_->GetFieldTypeStr();
781 }
782
783 private:
784 std::shared_ptr<const ChannelIndex> index_;
785 FieldPtr origin_field_;
786 };
787
788 public:
789 CombiningMergeActionResultImpl(BranchViewPtr left, BranchViewPtr right);
790
791 std::vector<std::string> GetFields() const override {
792 return field_names_;
793 }
794 size_t GetNumberOfChannels() const override {
795 return left_->GetNumberOfChannels() * right_->GetNumberOfChannels();
796 }
797 FieldPtr GetFieldPtr(std::string field_name) const override {
798 return fields_.at(field_name);
799 }
800 void SetEntry(Long64_t entry) const override {
801 left_->SetEntry(entry);
802 right_->SetEntry(entry);
803 left_index_->UpdateIndex(left_->GetNumberOfChannels(), right_->GetNumberOfChannels());
804 right_index_->UpdateIndex(left_->GetNumberOfChannels(), right_->GetNumberOfChannels());
805 }
806 Long64_t GetEntries() const override {
807 return left_->GetEntries();
808 }
809 BranchViewPtr Clone() const override {
810 return std::make_shared<CombiningMergeActionResultImpl>(left_->Clone(), right_->Clone());
811 }
812
813 private:
