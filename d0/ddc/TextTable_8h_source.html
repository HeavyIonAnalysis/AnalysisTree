192  TextTable::Row const & row = * rowIterator;
193  stream << table.vertical();
194  for ( unsigned i = 0; i < row.size(); ++i ) {
195  auto alignment = table.alignment( i ) == TextTable::Alignment::LEFT ? std::left : std::right;
196  // std::setw( width ) works as follows: a string which goes in the stream with byte length (!) l is filled with n spaces so that l+n=width.
197  // For a utf8 encoded string the glyph length g might be smaller than l. We need n spaces so that g+n=width which is equivalent to g+n+l-l=width ==> l+n = width+l-g
198  // l-g (that means glyph length minus byte length) has to be added to the width argument.
199  // l-g is computed by correctDistance.
200  stream << std::setw( table.width( i ) + table.correctDistance(row[ i ])) << alignment << row[ i ];
201  stream << table.vertical();
202  }
203  stream << "\n";
